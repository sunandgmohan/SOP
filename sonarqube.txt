environment {
        HARBOR_URL = "172.18.0.1"
        HARBOR_CREDS = "harbor-creds"
        PROJECT_NAME = "crm-backend"
        SONAR_HOME= tool 'sonar'
        IMAGE_NAME = 'gateway'
        IMAGE_TAG =  'latest'
        }
stage ('SonarQube quality analysis'){
          steps {
                withSonarQubeEnv("sonar") {
                    sh "$SONAR_HOME/bin/sonar-scanner -Dsonar.projectName=CRM-Backend -Dsonar.projectKey=CRM-Backend"
                      }
                   }
                 }
