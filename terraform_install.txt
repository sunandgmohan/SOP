🔨 installation

🧩 wget -O - https://apt.releases.hashicorp.com/gpg | sudo gpg --dearmor -o /usr/share/keyrings/hashicorp-archive-keyring.gpg
echo "deb [arch=$(dpkg --print-architecture) signed-by=/usr/share/keyrings/hashicorp-archive-keyring.gpg] https://apt.releases.hashicorp.com $(grep -oP '(?<=UBUNTU_CODENAME=).*' /etc/os-release || lsb_release -cs) main" | sudo tee /etc/apt/sources.list.d/hashicorp.list
sudo apt update && sudo apt install terraform


🧩 STEPS

🔨 Install awscli to sign in to aws from local matchine

📌  curl "https://awscli.amazonaws.com/awscli-exe-linux-x86_64.zip" -o "awscliv2.zip"
📌  unzip awscliv2.zip
📌  sudo ./aws/install

📌  aws config -- add access key and login

📌  Create a directory for your Terraform project and create a Terraform configuration file (usually named main.tf) in that directory. In this file, you define the AWS provider and resources you want to create. Here's a basic example

📌  got to this link to get the latest provider version and templets for .tf file creation
  🌐 https://registry.terraform.io/providers/hashicorp/aws/latest 

   terraform {
  required_providers {
    aws = {
      source = "hashicorp/aws"
      version = "5.95.0"
    }
  }
}

provider "aws" {
  region     = "ap-southeast-1"
}

# MAIN VPC
resource "aws_vpc" "my-vpc" {
  cidr_block       = "10.0.0.0/16"
  instance_tenancy = "default"

  tags = {
    Name = "Terra-VPC"
  }
}

# PUBLIC SUBNET 
resource "aws_subnet" "pubsub" {
  vpc_id     = aws_vpc.my-vpc.id
  cidr_block = "10.0.1.0/24"

  tags = {
    Name = "Terra-Public-Subnet"
  }
}

📌  terraform init

📌  terraform plan (to do a dry run)

📌  terraform apply

📌 cat terraform.tfstate (to see all the services created)

📌  terraform destroy (to remove)

📌  terraform init -upgrade (to upgrade to latest version)
