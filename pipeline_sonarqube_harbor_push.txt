pipeline {
    agent any

    environment {
        HARBOR_URL = "172.18.0.1"
        HARBOR_CREDS = "harbor-creds"
        PROJECT_NAME = "crm-backend"
        SONAR_HOME= tool 'sonar'
        IMAGE_NAME = 'gateway'
        IMAGE_TAG =  'latest'
        
        
        
       }

    stages {
        stage ('git checkout'){
          steps {
                git branch: 'main', credentialsId: 'gitkey', url: 'https://github.com/Client-Management-Project/CRM-Backend.git'
                  }
                }
        stage ('SonarQube quality analysis'){
          steps {
                withSonarQubeEnv("sonar") {
                    sh "$SONAR_HOME/bin/sonar-scanner -Dsonar.projectName=CRM-Backend -Dsonar.projectKey=CRM-Backend"
                      }
                   }
                 }
        stage('Quality Gate') {
            steps {
                timeout(time: 5, unit: 'MINUTES') {
                    waitForQualityGate abortPipeline: true
                }
            }
          }

        stage('build docker image'){
            steps{
                dir('gatewayservice') {
                    sh """
                     
                       docker build -t ${IMAGE_NAME}:${IMAGE_TAG} .
                     """
                   }
                 }
               }
        stage('Tag Docker Image') {
            steps {
              sh """
               docker tag ${IMAGE_NAME}:${IMAGE_TAG} \
                ${HARBOR_URL}/${PROJECT_NAME}/${IMAGE_NAME}:${IMAGE_TAG}
               """
               }
              }

        stage('Login to Harbor') {
          steps {
           withCredentials([usernamePassword(
           credentialsId: "${HARBOR_CREDS}",
           usernameVariable: 'HARBOR_USER',
           passwordVariable: 'HARBOR_PASS'
           )]) {

           sh """
            echo "$HARBOR_PASS" | docker login ${HARBOR_URL} \
            -u "$HARBOR_USER" --password-stdin
             """
             }
            }
           }

       stage('Push Image to Harbor') {
        steps {
         sh """
          docker push \
           ${HARBOR_URL}/${PROJECT_NAME}/${IMAGE_NAME}:${IMAGE_TAG}
           """
            }
          }
       
     }
   }
