📌 CLOUDFRONT DEPLOYMENT 

pipeline{
  agent any
 
    environment{
    S3_BUCKET='crm-managerpanel-qa'
    DISTRIBUTION_ID='E17J7998BMK9UK'
       }
 
   stages{
       stage('checkout'){
         steps{
          git branch: 'sunand', credentialsId: 'github-cred', url: 'https://github.com/Client-Management-Project/CRM-Frontend-ManagerPanel.git'
            }
           }
 
       stage('install dependendies'){
         steps{
          sh "npm install"
             }
           }
 
       stage('build static files'){
         steps{
          sh "npm run build"
              }
           }
 
       stage('move static file to s3 bucket'){
         steps{
           withCredentials([[$class: 'AmazonWebServicesCredentialsBinding', credentialsId: 'kiaq-aws-cred']]){
           sh "/usr/local/bin/aws s3 sync dist/ s3://${S3_BUCKET} --delete"
              }
             }
           }
 
       stage('invalidate cloudfront cache'){
         steps{
             withCredentials([[$class: 'AmazonWebServicesCredentialsBinding', credentialsId: 'kiaq-aws-cred']]){
             sh "/usr/local/bin/aws cloudfront create-invalidation --distribution-id ${DISTRIBUTION_ID} --paths '/*'"
               }
              }
             }
       }
      }



📌 backend DEPLOYMENT 


pipeline {
    agent any

    environment {
        IMAGE_NAME = 'app-backend'
        CONTAINER_NAME = 'backend-contain'
        EC2_IP = '54.197.113.122'
    }

    stages {

        stage('Checkout') {
            steps {
                git branch: 'main',
                    url: 'https://github.com/sunandgmohan/E-flow-Backend.git'
            }
        }

        stage('Deploy to EC2') {
            steps {
                sshagent(['EC2_CRED']) {
                    sh """
                          ssh -o StrictHostKeyChecking=no ubuntu@${EC2_IP} "
                          rm -rf /home/ubuntu/backend
                          mkdir -p /home/ubuntu/backend
                          cd /home/ubuntu/backend
                        

                          if [ ! -d E-flow-Backend ]; then 
                            git clone https://github.com/sunandgmohan/E-flow-Backend.git
                          fi 
                          
                           cd E-flow-Backend
                           git checkout main
                           git pull origin main
                        
                           
                           docker stop ${CONTAINER_NAME} || true
                           docker rm ${CONTAINER_NAME} || true
                           docker rmi ${IMAGE_NAME} || true
                           docker build -t ${IMAGE_NAME} .
                           docker run -d -p 3000:3000 --name ${CONTAINER_NAME} ${IMAGE_NAME}
                        "
                    """
                }
            }
        }
    }
}
